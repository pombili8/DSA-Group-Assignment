STRUCT Contact
    name: STRING
    phoneNumber: STRING
END STRUCT

phonebook = []

FUNCTION insertContact(name, phoneNumber)
    newContact = Contact(name, phoneNumber)
    ADD newContact TO phonebook
    PRINT "Contact added."
END FUNCTION

FUNCTION searchContact(name)
    FOR EACH contact IN phonebook
        IF contact.name == name THEN
            PRINT "Contact found: " + contact.name + " - " + contact.phoneNumber
            RETURN
        END IF
    END FOR
    PRINT "Contact not found."
END FUNCTION

FUNCTION deleteContact(name)
    FOR EACH contact IN phonebook
        IF contact.name == name THEN
            REMOVE contact FROM phonebook
            PRINT "Contact deleted."
            RETURN
        END IF
    END FOR
    PRINT "Contact not found."
END FUNCTION

FUNCTION displayContacts()
    IF phonebook IS EMPTY THEN
        PRINT "No contacts available."
    ELSE
        FOR EACH contact IN phonebook
            PRINT contact.name + " - " + contact.phoneNumber
        END FOR
    END IF
END FUNCTION
